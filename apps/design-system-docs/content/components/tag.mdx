import { Tag, Avatar, AvatarImage, AvatarFallback } from '@marmoui/ui';
import { MdStar, MdLabel, MdPerson } from 'react-icons/md';

<Accordions>
  <Accordion title="Overview">
    Tags are compact elements used to label, categorize, or group content. They support optional avatars, icons, and a close/delete button. Tags are **not clickable** — they display information and can optionally be removed.

    **When to Use:**
    - Displaying categories or labels on content items
    - Showing applied filters that can be removed
    - Grouping items (contacts, products, categories)
    - Tagging items in lists and tables

    **When NOT to Use:**
    - For status indicators (use Badge)
    - For navigation (use links)
    - For actions (use Button)
    - For clickable pills (use SegmentedControl or ToggleGroup)
  </Accordion>
</Accordions>

<Installation />


## Best Practices

<GuidelinesGrid>
	<Guideline type="do" title="Keep labels short">
		Tag text should be 1-3 words. Long text defeats the compact purpose.
	</Guideline>
	<Guideline type="do" title="Use closable for filters">
		When tags represent applied filters or selections, make them closable so users can remove them.
	</Guideline>
	<Guideline type="dont" title="Don't use as buttons">
		Tags display information. For clickable actions, use Button or Link.
	</Guideline>
	<Guideline type="dont" title="Don't overload a view with tags">
		Limit visible tags per item to 3-5. Show a "+N more" overflow for additional tags.
	</Guideline>
</GuidelinesGrid>

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `size` | `'sm' \| 'md' \| 'lg'` | `'lg'` | Tag size |
| `startElement` | <TypeInfo>ReactNode</TypeInfo> | - | Icon, avatar, or element before the text |
| `endElement` | <TypeInfo>ReactNode</TypeInfo> | - | Icon or element after the text (hidden when closable) |
| `closable` | `boolean` | `false` | Show a close/remove button |
| `onClose` | `() => void` | - | Callback when close button is clicked |
| `disabled` | `boolean` | `false` | Disable the tag and close button |
| `children` | <TypeInfo>ReactNode</TypeInfo> | - | Tag text content |
| `className` | `string` | - | Additional className |

## Accessibility

- Close button has `aria-label="Remove"` for screen readers.
- Close button is keyboard accessible with Enter/Space.
- Disabled state prevents interaction on the close button.
- Use descriptive text that makes sense without visual context.
